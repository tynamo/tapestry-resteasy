26  	public static void bind(ServiceBinder binder)
27  	{
28  		// Make bind() calls on the binder object to define most IoC services.
29  		// Use service builder methods (example below) when the implementation
30  		// is provided inline, or requires more initialization than simply
31  		// invoking the constructor.
32  		binder.bind(javax.ws.rs.core.Application.class, org.tynamo.resteasy.Application.class);
33  		binder.bind(HttpServletRequestFilter.class, ResteasyRequestFilter.class).withId("ResteasyRequestFilter");
34  		binder.bind(HttpServletRequestFilter.class, JSAPIRequestFilter.class).withId("JSAPIRequestFilter");
35  	}

37  	@Contribute(HttpServletRequestHandler.class)
38  	public static void httpServletRequestHandler(OrderedConfiguration<HttpServletRequestFilter> configuration,
39  	                                             @InjectService("ResteasyRequestFilter") HttpServletRequestFilter resteasyRequestFilter,
40  	                                             @InjectService("JSAPIRequestFilter") HttpServletRequestFilter jsapiRequestFilter)
41  	{
42  		configuration.add("ResteasyRequestFilter", resteasyRequestFilter, "after:IgnoredPaths", "before:GZIP");
43  		configuration.add("JSAPIRequestFilter", jsapiRequestFilter, "after:ResteasyRequestFilter");
44  	}

46  	@Contribute(SymbolProvider.class)
47  	@FactoryDefaults
48  	public static void setupSymbols(MappedConfiguration<String, Object> configuration)
49  	{
50  		configuration.add(ResteasySymbols.MAPPING_PREFIX, "/rest");
51  		configuration.add(ResteasySymbols.MAPPING_PREFIX_JSAPI, "/jsapi");
52  		configuration.add(ResteasySymbols.AUTOSCAN_REST_PACKAGE, true);
53  		configuration.add(ResteasySymbols.CORS_ENABLED, false);
54  	}

56  	@Contribute(javax.ws.rs.core.Application.class)
57  	public static void javaxWsRsCoreApplication(Configuration<Object> singletons,
58  	                                            ObjectLocator locator,
59  	                                            ResteasyPackageManager resteasyPackageManager,
60  	                                            ClassNameLocator classNameLocator,
61  	                                            Logger logger)
62  	{
63  		ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
64  
65  		for (String packageName : resteasyPackageManager.getPackageNames())
66  		{
67  			for (String className : classNameLocator.locateClassNames(packageName))
68  			{
69  				try
70  				{
71  					Class clazz = contextClassLoader.loadClass(className);
72  					Class rootResourceClass = GetRestful.getRootResourceClass(clazz);
73  
74  					if (rootResourceClass != null)
75  					{
76  						if (rootResourceClass.equals(clazz))
77  						{
78  							if (!clazz.isInterface())
79  							{
80  								singletons.add(locator.autobuild(clazz));
81  							}
82  						} else
83  						{
84  							try
85  							{
86  								singletons.add(locator.getService(rootResourceClass));
87  							} catch (RuntimeException e)
88  							{
89  								logger.info(e.getMessage());
90  								logger.info("Trying to create a proxy for " + rootResourceClass.getName());
91  								singletons.add(locator.proxy(rootResourceClass, clazz));
92  							}
93  						}
94  					} else if (clazz.isAnnotationPresent(Provider.class))
95  					{
96  						singletons.add(locator.autobuild(clazz));
97  					}
98  
99  				} catch (ClassNotFoundException ex)
100 				{
101 					throw new RuntimeException(ex);
102 				}
103 			}
104 		}
105 	}

107 	/**
108 	 * Contributes the package "&lt;root&gt;.rest" (InternalConstants.TAPESTRY_APP_PACKAGE_PARAM + ".rest")
109 	 * to the configuration, so that it will be scanned for annotated REST resource classes.
110 	 */
111 	@Contribute(ResteasyPackageManager.class)
112 	public static void resteasyPackageManager(Configuration<String> configuration,
113 	                                          @Symbol(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM) String appRootPackage,
114 	                                          @Symbol(ResteasySymbols.AUTOSCAN_REST_PACKAGE) Boolean shouldScanRestPackage)
115 	{
116 		if (shouldScanRestPackage)
117 		{
118 			configuration.add(appRootPackage + ".rest");
119 		}
120 	}
