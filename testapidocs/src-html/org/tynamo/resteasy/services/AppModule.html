001package org.tynamo.resteasy.services;
002
003import org.apache.tapestry5.internal.InternalConstants;
004import org.apache.tapestry5.internal.InternalSymbols;
005import org.apache.tapestry5.ioc.Configuration;
006import org.apache.tapestry5.ioc.MappedConfiguration;
007import org.apache.tapestry5.ioc.ServiceBinder;
008import org.apache.tapestry5.ioc.annotations.Contribute;
009import org.apache.tapestry5.ioc.annotations.ImportModule;
010import org.apache.tapestry5.ioc.services.ApplicationDefaults;
011import org.apache.tapestry5.ioc.services.SymbolProvider;
012import org.tynamo.resteasy.ResteasyModule;
013import org.tynamo.resteasy.ResteasyPackageManager;
014import org.tynamo.resteasy.ResteasySymbols;
015import org.tynamo.resteasy.rest.AutodiscoverableInjectableResource;
016import org.tynamo.resteasy.ws.ReloadableEchoResource;
017import org.tynamo.resteasy.ws.ReloadableEchoResourceImpl;
018
019@ImportModule(ResteasyModule.class)
020public class AppModule
021{
022
023        public static void bind(ServiceBinder binder)
024        {
025                binder.bind(ReloadableEchoResource.class, ReloadableEchoResourceImpl.class);
026                binder.bind(AutodiscoverableInjectableResource.class);
027        }
028
029
030        /**
031         * Contributions to the RESTeasy main Application, insert all your RESTeasy singleton services here.
032         * 
033         * @param singletons
034         *          jax-rs singleton resources
035         * @param reloadableEchoResource
036         *          T5 service to be contributed as a REST resource
037         * 
038         */
039        @Contribute(javax.ws.rs.core.Application.class)
040        public static void contributeApplication(Configuration<Object> singletons,
041                                                 ReloadableEchoResource reloadableEchoResource)
042        {
043                singletons.add(reloadableEchoResource);
044        }
045
046        @Contribute(SymbolProvider.class)
047        @ApplicationDefaults
048        public static void provideSymbols(MappedConfiguration<String, String> configuration)
049        {
050                configuration.add(ResteasySymbols.MAPPING_PREFIX, "/mycustomresteasyprefix");
051                configuration.add(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM, "org.tynamo.resteasy");
052                configuration.add(InternalSymbols.APP_PACKAGE_PATH, "org/tynamo/resteasy");
053        }
054
055        @Contribute(ResteasyPackageManager.class)
056        public static void resteasyPackageManager(Configuration<String> configuration)
057        {
058                configuration.add("org.tynamo.resteasy.ws.autobuild");
059        }
060
061}
