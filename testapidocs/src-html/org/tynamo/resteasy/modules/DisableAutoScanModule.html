001package org.tynamo.resteasy.modules;
002
003import org.apache.tapestry5.ioc.MappedConfiguration;
004import org.apache.tapestry5.ioc.annotations.Contribute;
005import org.apache.tapestry5.ioc.services.ApplicationDefaults;
006import org.apache.tapestry5.ioc.services.SymbolProvider;
007import org.tynamo.resteasy.ResteasySymbols;
008
009public class DisableAutoScanModule
010{
011        @Contribute(SymbolProvider.class)
012        @ApplicationDefaults
013        public static void provideSymbols(MappedConfiguration<String, String> configuration)
014        {
015                configuration.add(ResteasySymbols.AUTOSCAN_REST_PACKAGE, "false");
016        }
017
018}
