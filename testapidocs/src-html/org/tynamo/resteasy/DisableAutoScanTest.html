001package org.tynamo.resteasy;
002
003import org.apache.tapestry5.ioc.Registry;
004import org.apache.tapestry5.ioc.RegistryBuilder;
005import org.apache.tapestry5.modules.TapestryModule;
006import org.apache.tapestry5.test.TapestryTestCase;
007import org.testng.Assert;
008import org.testng.annotations.Test;
009import org.tynamo.resteasy.services.AppModule;
010import org.tynamo.resteasy.modules.DisableAutoScanModule;
011
012public class DisableAutoScanTest extends TapestryTestCase
013{
014
015        @Test
016        public void auto_scan_enabled()
017        {
018                RegistryBuilder builder = new RegistryBuilder();
019
020                builder.add(TapestryModule.class);
021                builder.add(ResteasyModule.class);
022                builder.add(AppModule.class);
023
024                Registry registry = builder.build();
025                registry.performRegistryStartup();
026
027                javax.ws.rs.core.Application app = registry.getService(javax.ws.rs.core.Application.class);
028                Assert.assertEquals(app.getSingletons().size(), 9, "there are five services");
029
030                registry.shutdown();
031
032        }
033
034        @Test
035        public void auto_scan_disabled()
036        {
037                RegistryBuilder builder = new RegistryBuilder();
038
039                builder.add(TapestryModule.class);
040                builder.add(ResteasyModule.class);
041                builder.add(AppModule.class);
042                builder.add(DisableAutoScanModule.class);
043
044
045                Registry registry = builder.build();
046                registry.performRegistryStartup();
047
048                javax.ws.rs.core.Application app = registry.getService(javax.ws.rs.core.Application.class);
049                Assert.assertEquals(app.getSingletons().size(), 6, "there are two services");
050
051                registry.shutdown();
052        }
053
054}
