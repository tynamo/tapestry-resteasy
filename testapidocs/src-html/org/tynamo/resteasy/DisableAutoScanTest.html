001package org.tynamo.resteasy;
002
003import org.apache.tapestry5.ioc.Registry;
004import org.apache.tapestry5.ioc.RegistryBuilder;
005import org.apache.tapestry5.modules.TapestryModule;
006import org.testng.Assert;
007import org.testng.annotations.Test;
008import org.tynamo.resteasy.modules.DisableAutoScanModule;
009import org.tynamo.resteasy.services.AppModule;
010
011public class DisableAutoScanTest
012{
013
014        @Test
015        public void auto_scan_enabled()
016        {
017                RegistryBuilder builder = new RegistryBuilder();
018
019                builder.add(TapestryModule.class);
020                builder.add(ResteasyModule.class);
021                builder.add(AppModule.class);
022
023                Registry registry = builder.build();
024                registry.performRegistryStartup();
025
026                javax.ws.rs.core.Application app = registry.getService(javax.ws.rs.core.Application.class);
027                Assert.assertEquals(app.getSingletons().size(), 5, "there are five services");
028
029                registry.shutdown();
030
031        }
032
033        @Test
034        public void auto_scan_disabled()
035        {
036                RegistryBuilder builder = new RegistryBuilder();
037
038                builder.add(TapestryModule.class);
039                builder.add(ResteasyModule.class);
040                builder.add(AppModule.class);
041                builder.add(DisableAutoScanModule.class);
042
043
044                Registry registry = builder.build();
045                registry.performRegistryStartup();
046
047                javax.ws.rs.core.Application app = registry.getService(javax.ws.rs.core.Application.class);
048                Assert.assertEquals(app.getSingletons().size(), 2, "there are two services");
049
050                registry.shutdown();
051        }
052
053}
