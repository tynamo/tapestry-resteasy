001package org.tynamo.resteasy;
002
003
004import org.apache.tapestry5.internal.InternalConstants;
005import org.apache.tapestry5.ioc.*;
006import org.apache.tapestry5.ioc.annotations.Contribute;
007import org.apache.tapestry5.ioc.annotations.InjectService;
008import org.apache.tapestry5.ioc.annotations.Symbol;
009import org.apache.tapestry5.ioc.services.ClassNameLocator;
010import org.apache.tapestry5.ioc.services.FactoryDefaults;
011import org.apache.tapestry5.ioc.services.SymbolProvider;
012import org.apache.tapestry5.services.HttpServletRequestFilter;
013import org.apache.tapestry5.services.HttpServletRequestHandler;
014import org.jboss.resteasy.util.GetRestful;
015import org.slf4j.Logger;
016
017import javax.ws.rs.ext.Provider;
018import java.util.Collection;
019
020/**
021 * This module is automatically included as part of the Tapestry IoC Registry
022 */
023public class ResteasyModule
024{
025
026        public static void bind(ServiceBinder binder)
027        {
028                // Make bind() calls on the binder object to define most IoC services.
029                // Use service builder methods (example below) when the implementation
030                // is provided inline, or requires more initialization than simply
031                // invoking the constructor.
032                binder.bind(javax.ws.rs.core.Application.class, org.tynamo.resteasy.Application.class);
033                binder.bind(HttpServletRequestFilter.class, ResteasyRequestFilter.class).withId("ResteasyRequestFilter");
034                binder.bind(HttpServletRequestFilter.class, JSAPIRequestFilter.class).withId("JSAPIRequestFilter");
035        }
036
037        @Contribute(HttpServletRequestHandler.class)
038        public static void httpServletRequestHandler(OrderedConfiguration<HttpServletRequestFilter> configuration,
039                                                     @InjectService("ResteasyRequestFilter") HttpServletRequestFilter resteasyRequestFilter,
040                                                     @InjectService("JSAPIRequestFilter") HttpServletRequestFilter jsapiRequestFilter)
041        {
042                configuration.add("ResteasyRequestFilter", resteasyRequestFilter, "after:IgnoredPaths", "before:GZIP");
043                configuration.add("JSAPIRequestFilter", jsapiRequestFilter, "after:ResteasyRequestFilter");
044        }
045
046        @Contribute(SymbolProvider.class)
047        @FactoryDefaults
048        public static void setupSymbols(MappedConfiguration<String, Object> configuration)
049        {
050                configuration.add(ResteasySymbols.MAPPING_PREFIX, "/rest");
051                configuration.add(ResteasySymbols.MAPPING_PREFIX_JSAPI, "/jsapi");
052                configuration.add(ResteasySymbols.AUTOSCAN_REST_PACKAGE, true);
053                configuration.add(ResteasySymbols.CORS_ENABLED, false);
054        }
055
056        @Contribute(javax.ws.rs.core.Application.class)
057        public static void javaxWsRsCoreApplication(Configuration<Object> singletons,
058                                                    ObjectLocator locator,
059                                                    ResteasyPackageManager resteasyPackageManager,
060                                                    ClassNameLocator classNameLocator,
061                                                    Logger logger)
062        {
063                ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
064
065                for (String packageName : resteasyPackageManager.getPackageNames())
066                {
067                        for (String className : classNameLocator.locateClassNames(packageName))
068                        {
069                                try
070                                {
071                                        Class clazz = contextClassLoader.loadClass(className);
072                                        Class rootResourceClass = GetRestful.getRootResourceClass(clazz);
073
074                                        if (rootResourceClass != null)
075                                        {
076                                                if (rootResourceClass.equals(clazz))
077                                                {
078                                                        if (!clazz.isInterface())
079                                                        {
080                                                                singletons.add(locator.autobuild(clazz));
081                                                        }
082                                                } else
083                                                {
084                                                        try
085                                                        {
086                                                                singletons.add(locator.getService(rootResourceClass));
087                                                        } catch (RuntimeException e)
088                                                        {
089                                                                logger.info(e.getMessage());
090                                                                logger.info("Trying to create a proxy for " + rootResourceClass.getName());
091                                                                singletons.add(locator.proxy(rootResourceClass, clazz));
092                                                        }
093                                                }
094                                        } else if (clazz.isAnnotationPresent(Provider.class))
095                                        {
096                                                singletons.add(locator.autobuild(clazz));
097                                        }
098
099                                } catch (ClassNotFoundException ex)
100                                {
101                                        throw new RuntimeException(ex);
102                                }
103                        }
104                }
105        }
106
107        /**
108         * Contributes the package "&lt;root&gt;.rest" (InternalConstants.TAPESTRY_APP_PACKAGE_PARAM + ".rest")
109         * to the configuration, so that it will be scanned for annotated REST resource classes.
110         */
111        @Contribute(ResteasyPackageManager.class)
112        public static void resteasyPackageManager(Configuration<String> configuration,
113                                                  @Symbol(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM) String appRootPackage,
114                                                  @Symbol(ResteasySymbols.AUTOSCAN_REST_PACKAGE) Boolean shouldScanRestPackage)
115        {
116                if (shouldScanRestPackage)
117                {
118                        configuration.add(appRootPackage + ".rest");
119                }
120        }
121
122        public static ResteasyPackageManager buildResteasyPackageManager(final Collection<String> packageNames)
123        {
124                return new ResteasyPackageManager()
125                {
126                        public Collection<String> getPackageNames()
127                        {
128                                return packageNames;
129                        }
130                };
131        }
132}
